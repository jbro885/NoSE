def pp_s(*objs)

  objs.each { |obj| PP.pp(obj, s) }
  s.rewind
  s.read
end

def suppress_warnings
  original_verbosity = $VERBOSE
  $VERBOSE = nil
  result = yield
  $VERBOSE = original_verbosity

  result
end
