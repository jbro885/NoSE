def validate_json(json)
  schema_file = File.join File.dirname(__FILE__), '..', '..',
                          'data', 'nose', 'nose-schema.json'
  schema = JSON.parse File.read(schema_file)

  data = JSON.parse json
  JSON::Validator.validate(schema, data)
end
