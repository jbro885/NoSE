def load(name)
  if File.exist? name
    filename = name
  else
    path = const_get(:LOAD_PATH)
    filename = File.expand_path "../../../#{path}/#{name}.rb", __FILE__
  end

  source_code = File.read(filename)
  instance = binding.eval source_code, filename
  instance.instance_variable_set :@source_code, source_code
  instance
end
